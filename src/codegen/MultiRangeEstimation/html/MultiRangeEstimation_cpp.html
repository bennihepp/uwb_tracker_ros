  519     }
  520   
  521     a[3] = params->UWB4p0;
  522     b[3] = params->UWB4p1;
  523   
  524     //  prediction
  525     for (k = 0; k < 6; k++) {
  526       n[k] = x[k];
  527     }
  528   
  529     for (i1 = 0; i1 < 6; i1++) {
  530       x[i1] = 0.0;
  531       for (i2 = 0; i2 < 6; i2++) {
  532         x[i1] += d_a[i1 + 6 * i2] * n[i2];
  533         c_a[i1 + 6 * i2] = 0.0;
  534         for (k = 0; k < 6; k++) {
  535           c_a[i1 + 6 * i2] += d_a[i1 + 6 * k] * P[k + 6 * i2];
  536         }
  537       }
  538     }
  539   
  540     for (k = 0; k < 6; k++) {
  541       for (i1 = 0; i1 < 6; i1++) {
  542         temp = 0.0;
  543         for (i2 = 0; i2 < 6; i2++) {
  544           temp += c_a[k + 6 * i2] * b_b[i2 + 6 * i1];
  545         }
  546   
  547         P[k + 6 * i1] = temp + varargin_1[k + 6 * i1];
  548       }
  549   
  550       n[k] = x[k];
  551     }
  552   
  553     for (k = 0; k < 3; k++) {
  554       pos[k] = x[k];
  555     }
  556   
  557     InitialRangeArray[0] = rangeArray_data[0];
  558     InitialRangeArray[1] = rangeArray_data[1];
  559     InitialRangeArray[2] = rangeArray_data[2];
  560     InitialRangeArray[3] = rangeArray_data[3];
  561     K_size_idx_0 = 4;
  562     K_size_idx_1 = 6;
  563     memset(&K_data[0], 0, 24U * sizeof(double));
  564     for (iter = 0; iter < 4; iter++) {
  565       for (cntUWB = 0; cntUWB < 4; cntUWB++) {
  566         //          a=[0;0;0;0];
  567         //          b=[1;1;1;1];
  568         for (i1 = 0; i1 < 3; i1++) {
  569           b_pos[i1] = pos[i1] - o[i1 + 3 * cntUWB];
  570         }
  571   
  572         h_n = norm(b_pos);
  573   
  574         // derivatives
  575         h_to_hn = 1.0 / b[cntUWB];
  576         hn_to_hs = 1.0 / (2.0 * sqrt(h_n));
  577         for (i1 = 0; i1 < 3; i1++) {
  578           H[cntUWB + (i1 << 2)] = h_to_hn * (hn_to_hs * (2.0 * pos[i1] - 2.0 *
  579             o[i1 + 3 * cntUWB]));
  580         }
  581   
  582         H[12 + cntUWB] = 0.0;
  583         H[16 + cntUWB] = 0.0;
  584         H[20 + cntUWB] = 0.0;
  585         h[cntUWB] = (-a[cntUWB] + h_n) / b[cntUWB];
  586         res[cntUWB] = InitialRangeArray[cntUWB] - h[cntUWB];
  587         for (i1 = 0; i1 < 6; i1++) {
  588           b_H[cntUWB + (i1 << 2)] = 0.0;
  589           for (i2 = 0; i2 < 6; i2++) {
  590             b_H[cntUWB + (i1 << 2)] += H[cntUWB + (i2 << 2)] * P[i2 + 6 * i1];
  591           }
  592         }
  593       }
  594   
  595       for (i1 = 0; i1 < 4; i1++) {
  596         for (i2 = 0; i2 < 4; i2++) {
  597           temp = 0.0;
  598           for (k = 0; k < 6; k++) {
  599             temp += b_H[i1 + (k << 2)] * H[i2 + (k << 2)];
  600           }
  601   
  602           S[i1 + (i2 << 2)] = temp + R[i1 + (i2 << 2)];
  603         }
  604       }
  605   
  606       for (i1 = 0; i1 < 6; i1++) {
  607         for (i2 = 0; i2 < 4; i2++) {
  608           dv0[i1 + 6 * i2] = 0.0;
  609           for (k = 0; k < 6; k++) {
  610             dv0[i1 + 6 * i2] += P[i1 + 6 * k] * H[i2 + (k << 2)];
  611           }
  612         }
  613       }
  614   
  615       mrdivide(dv0, S);
  616       K_size_idx_0 = 6;
  617       K_size_idx_1 = 4;
  618       memcpy(&K_data[0], &dv0[0], 24U * sizeof(double));
  619   
  620       //  chi^2 outlier rejection
  621       xgetrf(S, ipiv, &k);
  622       for (i1 = 0; i1 < 4; i1++) {
  623         y[i1] = res[i1];
  624       }
  625   
  626       for (cr = 0; cr < 4; cr++) {
  627         br = cr << 2;
  628         for (k = 0; k + 1 <= cr; k++) {
  629           if (S[k + br] != 0.0) {
  630             y[cr] -= S[k + br] * y[k];
  631           }
  632         }
  633   
  634         y[cr] *= 1.0 / S[cr + br];
  635       }
  636   
  637       for (cr = 3; cr >= 0; cr += -1) {
  638         br = cr << 2;
  639         for (k = cr + 1; k + 1 < 5; k++) {
  640           if (S[k + br] != 0.0) {
  641             y[cr] -= S[k + br] * y[k];
  642           }
  643         }
  644       }
  645   
  646       for (k = 2; k >= 0; k += -1) {
  647         if (ipiv[k] != k + 1) {
  648           temp = y[k];
  649           y[k] = y[ipiv[k] - 1];
  650           y[ipiv[k] - 1] = temp;
  651         }
  652       }
  653   
  654       temp = 0.0;
  655       for (i1 = 0; i1 < 4; i1++) {
  656         temp += y[i1] * res[i1];
  657       }
  658   
  659       if (sqrt(temp) < 4.0) {
  660         for (i1 = 0; i1 < 6; i1++) {
  661           C_data[i1] = x[i1] - n[i1];
  662         }
  663   
  664         for (i1 = 0; i1 < 4; i1++) {
  665           temp = 0.0;
  666           for (i2 = 0; i2 < 6; i2++) {
  667             temp += H[i1 + (i2 << 2)] * C_data[i2];
  668           }
  669   
  670           y[i1] = res[i1] - temp;
  671         }
  672   
  673         for (i1 = 0; i1 < 6; i1++) {
  674           C_data[i1] = 0.0;
  675         }
  676   
  677         for (ic = 1; ic < 7; ic++) {
  678           C_data[ic - 1] = 0.0;
  679         }
  680   
  681         ar = 0;
  682         for (ib = 0; ib + 1 < 5; ib++) {
  683           if (y[ib] != 0.0) {
  684             ia = ar;
  685             for (ic = 0; ic + 1 < 7; ic++) {
  686               ia++;
  687               C_data[ic] += y[ib] * K_data[ia - 1];
  688             }
  689           }
  690   
  691           ar += 6;
  692         }
  693   
  694         for (i1 = 0; i1 < 6; i1++) {
  695           n[i1] = x[i1] + C_data[i1];
  696         }
  697   
  698         outl = 0;
  699       } else {
  700         outl = 1;
  701       }
  702     }
  703   
  704     if (outl == 0) {
  705       //  only do update if there was no outlyer
  706       for (i1 = 0; i1 < 6; i1++) {
  707         for (i2 = 0; i2 < K_size_idx_0; i2++) {
  708           b_C_data[i2 + K_size_idx_0 * i1] = 0.0;
  709         }
  710       }
  711   
  712       k = K_size_idx_0 * 5;
  713       for (cr = 0; cr <= k; cr += K_size_idx_0) {
  714         i1 = cr + K_size_idx_0;
  715         for (ic = cr; ic + 1 <= i1; ic++) {
  716           b_C_data[ic] = 0.0;
  717         }
  718       }
  719   
  720       br = 0;
  721       for (cr = 0; cr <= k; cr += K_size_idx_0) {
  722         ar = 0;
  723         i1 = br + K_size_idx_1;
  724         for (ib = br; ib + 1 <= i1; ib++) {
  725           if (H[ib] != 0.0) {
  726             ia = ar;
  727             i2 = cr + K_size_idx_0;
  728             for (ic = cr; ic + 1 <= i2; ic++) {
  729               ia++;
  730               b_C_data[ic] += H[ib] * K_data[ia - 1];
  731             }
  732           }
  733   
  734           ar += K_size_idx_0;
  735         }
  736   
  737         br += K_size_idx_1;
  738       }
  739   
  740       for (i1 = 0; i1 < 36; i1++) {
  741         I[i1] = 0;
  742       }
  743   
  744       for (k = 0; k < 6; k++) {
  745         I[k + 6 * k] = 1;
  746       }
  747   
  748       memcpy(&B[0], &P[0], 36U * sizeof(double));
  749       for (i1 = 0; i1 < 6; i1++) {
  750         for (i2 = 0; i2 < 6; i2++) {
  751           c_a[i2 + 6 * i1] = (double)I[i2 + 6 * i1] - b_C_data[i2 + 6 * i1];
  752         }
  753       }
  754   
  755       for (i1 = 0; i1 < 6; i1++) {
  756         for (i2 = 0; i2 < 6; i2++) {
  757           P[i1 + 6 * i2] = 0.0;
  758           for (k = 0; k < 6; k++) {
  759             P[i1 + 6 * i2] += c_a[i1 + 6 * k] * B[k + 6 * i2];
  760           }
  761         }
  762       }
  763     }
  764   
  765     //  copy the new state n after the iterated update into the real state
  766     // % state augmentation
  767     //  find first oldest state
  768     //  copy the state to the output variables
  769     for (k = 0; k < 6; k++) {
  770       x[k] = n[k];
  771       x_out[k] = x[k];
  772     }
